my-restaurant-app/
├── .gitignore            # (Git 무시 파일 - 생성 추천)
├── index.html            # Vite 앱의 진입점이 되는 HTML 쉘
├── package.json          # 프로젝트 의존성 (vue, pinia, ol, tailwindcss 등)
├── tailwind.config.js    # Tailwind CSS 설정 (스캔할 파일 경로 지정)
├── postcss.config.js     # PostCSS 설정 (Tailwind, Autoprefixer 플러그인)
│
└── src/                  # 애플리케이션 소스 코드
│
├── api/
│   └── mockApi.js      # (Model) 백엔드 API 시뮬레이션
│
├── assets/             # (정적 에셋. 예: CSS, 이미지 등)
│   └── (현재 비어있음)
│
├── components/         # (View) 재사용 가능한 UI 컴포넌트
│   ├── MapComponent.vue  # OpenLayers 지도 뷰
│   └── Sidebar.vue       # 맛집 정보 및 댓글 사이드바 뷰
│
├── router/
│   └── index.js        # (ViewModel) Vue Router 설정 (페이지 이동, 네비게이션 가드)
│
├── store/
│   ├── auth.js           # (Model) Pinia 'auth' 스토어 (로그인/로그아웃 상태, 유저 정보)
│   └── restaurants.js    # (Model) Pinia 'restaurants' 스토어 (맛집 목록, 댓글)
│
├── views/              # (View) 페이지 단위의 메인 컴포넌트
│   ├── HomeView.vue      # 메인 페이지 (지도 + 사이드바 레이아웃)
│   └── LoginView.vue     # 로그인 페이지
│
├── App.vue             # (VM) 최상위 루트 컴포넌트 (router-view 래퍼, 전역 CSS)
└── main.js             # 앱의 메인 진입점 (Vue 앱 생성, Pinia/Router 플러그인 등록)
